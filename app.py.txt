    page_title="Cybersecurity Portfolio",
    page_icon="🛡️",
    layout="wide"
)

Welcome to my cybersecurity portfolio. This portfolio presents my academic and practical projects 
in cyber forensics, IT auditing, vulnerability analysis, penetration testing, and security research.
""")

# Resume Section
st.header("📌 Resume")

resume_path = "Alaa Al Saad_Cyber Forensics & Security_Resume.pdf"

if os.path.exists(resume_path):
    with open(resume_path, "rb") as file:
        st.download_button(
            label="Download My Resume",
            data=file,
            file_name="Alaa Al Saad_Cyber Forensics & Security_Resume.pdf",
            mime="application/pdf"
        )
else:
    st.warning("Resume file not found. Please make sure the PDF is in the same folder as app.py.")

# Project folders
project_folders = {
    "Cyber Forensics": "projects/Cyber_Forensics",
    "Information Technology Auditing": "projects/Information_Technology_Auditing",
    "Vulnerability Analysis and Control": "projects/Vulnerability_Analysis_and_Control",
    "MIS Project": "projects/MIS_Project"
}

st.header("📂 Cybersecurity Projects")

for project_name, folder_path in project_folders.items():
    st.subheader(project_name)

    if os.path.exists(folder_path):
        files = os.listdir(folder_path)

        if files:
            for file_name in files:
                file_path = os.path.join(folder_path, file_name)

                with st.expander(file_name):
                    st.write("This file is part of my cybersecurity academic and practical project work.")

                    with open(file_path, "rb") as file:
                        st.download_button(
                            label=f"Download {file_name}",
                            data=file,
                            file_name=file_name,
                            key=file_path
                        )
        else:
            st.info("No files added yet.")
    else:
        st.warning(f"Folder not found: {folder_path}")
